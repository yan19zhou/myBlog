#### provide 和 inject 组件跨级拿到数据
    //祖先组件
    provide(){
        return {
            value:this.value,
            zuxian:this
        }
    },
    data(){
      return {value:"aaa"}  
    }
    // 后代组件
    inject:['value','zuxian'],
    mounted(){
        console.log(this.value) // aaa
    }
